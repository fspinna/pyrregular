def infer_static_columns(df, id_column, dropna=False):
    df_grouped = df.groupby(id_column).nunique(dropna=dropna)
    return [
        c
        for c in df_grouped.columns
        if np.array_equal(df_grouped[c].to_numpy(), np.ones(len(df_grouped[c])))
        and c != id_column
    ]
